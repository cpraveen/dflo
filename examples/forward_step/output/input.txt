subsection limiter

  set M                         = 0

  # limiter parameter
  set beta                      = 1.0

  # whether to use characteristic limiter
  set characteristic limiter    = true    # default: false

  # conserve angular momentum during limiting
  set conserve angular momentum = false

  # whether to use positivity limiter
  set positivity limiter        = true    # default: false

  # Shock indicator type: limiter | density | energy | u2
  set shock indicator           = limiter

  # Limiter type: none | TVB
  set type                      = TVB     # default: none
end


subsection linear solver

  set ilut absolute tolerance = 1e-9

  # Ilut drop tolerance
  set ilut drop tolerance     = 1e-10

  # Ilut preconditioner fill
  set ilut fill               = 2

  # Ilut relative tolerance
  set ilut relative tolerance = 1.1

  # Maximum solver iterations
  set max iters               = 300

  # The kind of solver for the linear system. Choices are
  # <gmres|direct|umfpack|rk3|mood>.
  set method                  = rk3

  # State whether output from solver runs should be printed. Choices are
  # <quiet|verbose>.
  set output                  = quiet

  # Linear solver residual
  set residual                = 1e-10
end


subsection output

  set compute angular momentum = 10000000

  # Output format for visualization: vtk, tecplot

  # Output once per this iteration period

  # Whether or not to produce schlieren plots
  set schlieren plot           = true     # default: false

  # Output once per this time period
  set time step                = 1e20
end
